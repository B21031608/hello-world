//自己实现字符串拷贝函数
char* StrCpy(char* src)
{
//1.计算src的长度 len
int len=strlen(src);
//2.用malloc申请长度为len+1的内存空间
char* Src=(char*)malloc((len+1)sizeof(char));
//3.将src字符串逐个字符复制到新申请的内存空间
while(src[0])
{
Src[0]=src[0];
src++;
}
//4.新申请内存空间的最后一位放入'\0'
Src[len]='\0';
//5.返回新申请的内存空间指针
return src;
}
